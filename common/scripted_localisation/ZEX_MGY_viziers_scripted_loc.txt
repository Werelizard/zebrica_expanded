# Displays the Vizier Value

defined_text = {
	name = ZEX_MGY_GET_VIZIER_KING_BALANCE
	text = {
		trigger = {
			check_variable = {
				ZEX_MGY_vizier_King_Balance < 10
			}
			check_variable = {
				ZEX_MGY_vizier_King_Balance > -10
			}
		}
		localization_key = ZEX_MGY_VIZIER_OFFICE_KING_BALANCED
	}
	text = {
		trigger = {
			check_variable = {
				ZEX_MGY_vizier_King_Balance < -66
			}
		}
		localization_key = ZEX_MGY_VIZIER_OFFICE_VIZIER_DOMINANT
	}
	text = {
		trigger = {
			check_variable = {
				ZEX_MGY_vizier_King_Balance < -33
			}
		}
		localization_key = ZEX_MGY_VIZIER_OFFICE_VIZIER_STRONG
	}
	text = {
		trigger = {
			check_variable = {
				ZEX_MGY_vizier_King_Balance < -10
			}
		}
		localization_key = ZEX_MGY_VIZIER_OFFICE_VIZIER_WEAK
	}
	text = {
		trigger = {
			check_variable = {
				ZEX_MGY_vizier_King_Balance > 66
			}
		}
		localization_key = ZEX_MGY_VIZIER_OFFICE_KING_DOMINANT
	}
	text = {
		trigger = {
			check_variable = {
				ZEX_MGY_vizier_King_Balance > 33
			}
		}
		localization_key = ZEX_MGY_VIZIER_OFFICE_KING_STRONG
	}
	text = {
		trigger = {
			check_variable = {
				ZEX_MGY_vizier_King_Balance > 10
			}
		}
		localization_key = ZEX_MGY_VIZIER_OFFICE_KING_WEAK
	}
}
