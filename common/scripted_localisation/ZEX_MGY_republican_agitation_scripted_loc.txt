#Sineighi

defined_text = {
	name = ZEX_MGY_GetSineighiAlignment
	text = {
		trigger = {
			check_variable = {
				var = ZEX_MGY_VAR_Sineighi_Alignment
				value = 8
				compare = greater_than_or_equals
			}
		}
		localization_key = ZEX_MGY_COMMUNIST_SINEIGHI
	}
	text = {
		trigger = {
			check_variable = {
				var = ZEX_MGY_VAR_Sineighi_Alignment
				value = 2
				compare = less_than_or_equals
			}
		}
		localization_key = ZEX_MGY_HARMONIST_SINEIGHI
	}
	text = {
		localization_key = ZEX_MGY_NEUTRAL_SINEIGHI
	}
}
