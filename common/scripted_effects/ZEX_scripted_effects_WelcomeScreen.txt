zex_open_splash_screen_start = {
	#This opens the splash screen, WHILST displaying the change / patchnotes for the current version
	set_variable = {
		zex_get_splashscreenstate_var = 1
	}
	set_country_flag = zex_splashscreen_show_patchnotes_flag
}

zex_open_splash_screen_ingame = {
	#This opens the splash screen, but does NOT display the change / patchnotes for the current version
	set_variable = {
		zex_get_splashscreenstate_var = 1
	}
}

zex_close_splash_screen = {
	#Closes the splashscreen and ensures that the patchnotes will never get in again
	set_variable = {
		zex_get_splashscreenstate_var = 0
	}
	if = {
		limit = {
			has_country_flag = zex_splashscreen_show_patchnotes_flag
		}
		clr_country_flag = zex_splashscreen_show_patchnotes_flag
	}
}

zex_splashscreen_show_tab_history = {
	# Shows the history tab
	set_variable = {
		zex_get_splashscreen_history_tab_var = 1
	}
	#Closes other tabs
	set_variable = {
		zex_get_splashscreen_characters_tab_var = 0
	}
	set_variable = {
		zex_get_splashscreen_paths_tab_var = 0
	}
	set_variable = {
		zex_get_splashscreen_credits_tab_var = 0
	}
}
