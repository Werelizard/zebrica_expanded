ZEX_MGY_Calculate_Total_Support = { #This is to reupdate the diet's support
	set_variable = {
		ZEX_MGY_Royal_Diet_Total_Support = 0
	}
	if = {
		limit = {
			check_variable = {
				ZEX_MGY_Royal_Diet_Zebrican_Loyalty > 70
			}
		}
		add_to_variable = {
			ZEX_MGY_Royal_Diet_Total_Support = ZEX_MGY_royal_diet_seat_array^0
		}
	}
	else_if = {
		limit = {
			check_variable = {
				ZEX_MGY_Royal_Diet_Zebrican_Loyalty > 30
			}
		}
			set_variable = {
				ZEX_MGY_Seats_Support_Temp = ZEX_MGY_royal_diet_seat_array^0
			}
			multiply_variable = {
				ZEX_MGY_Seats_Support_Temp = 0.5
			}
			round_variable = ZEX_MGY_Seats_Support_Temp
			add_to_variable = {
			ZEX_MGY_Royal_Diet_Total_Support = ZEX_MGY_Seats_Support_Temp
		}
	}
	if = {
		limit = {
			check_variable = {
				ZEX_MGY_Royal_Diet_Harmonist_Loyalty > 70
			}
		}
		add_to_variable = {
			ZEX_MGY_Royal_Diet_Total_Support = ZEX_MGY_royal_diet_seat_array^1
		}
	}
	else_if = {
		limit = {
			check_variable = {
				ZEX_MGY_Royal_Diet_Harmonist_Loyalty > 30
			}
		}
			set_variable = {
				ZEX_MGY_Seats_Support_Temp = ZEX_MGY_royal_diet_seat_array^1
			}
			multiply_variable = {
				ZEX_MGY_Seats_Support_Temp = 0.5
			}
			round_variable = ZEX_MGY_Seats_Support_Temp
			add_to_variable = {
			ZEX_MGY_Royal_Diet_Total_Support = ZEX_MGY_Seats_Support_Temp
		}
	}
	if = {
		limit = {
			check_variable = {
				ZEX_MGY_Royal_Diet_Loyalists_Loyalty > 70
			}
		}
		add_to_variable = {
			ZEX_MGY_Royal_Diet_Total_Support = ZEX_MGY_royal_diet_seat_array^2
		}
	}
	else_if = {
		limit = {
			check_variable = {
				ZEX_MGY_Royal_Diet_Loyalists_Loyalty > 30
			}
		}
			set_variable = {
				ZEX_MGY_Seats_Support_Temp = ZEX_MGY_royal_diet_seat_array^2
			}
			multiply_variable = {

ZEX_MGY_Diet_Reload_GUI = {
	clear_array = parliament_main_array
	for_each_loop = {
		array = ZEX_MGY_royal_diet_seat_array

		add_to_array = { temp_seat = ZEX_MGY_royal_diet_seat_array^i }
	}

	set_temp_variable = { temp_index_1 = 0 }
	set_temp_variable = { temp_index_2 = 1 }

	set_temp_variable = { temp_size = temp_seat^num }
	subtract_from_temp_variable = { temp_size = 1 }

	while_loop_effect = {
		limit = {
			check_variable = { temp_index_1 < temp_size }
		}
		
		add_to_variable = { temp_seat^temp_index_2 = temp_seat^temp_index_1 }
		add_to_temp_variable = { temp_index_1 = 1 }
		add_to_temp_variable = { temp_index_2 = 1 }
	}

	for_loop_effect = {
		start = 0
		end = temp_seat^num

		resize_array = { array = parliament_main_array value = parliament_member_array^v size = temp_seat^v }
	}

	clear_array = temp_seat
}

ZEX_MGY_Diet_modify_parliament_seat = {
	subtract_from_temp_variable = { temp_member = 1 }
	set_temp_variable = { temp = ZEX_MGY_royal_diet_seat_array^temp_member }
	add_to_temp_variable = { temp = ZEX_MGY_royal_diet_temp_seats }

	if = {
		limit = {
			check_variable = { temp > parliament_main_array^num }
		}
		subtract_from_temp_variable = { temp = parliament_main_array^num }
		subtract_from_temp_variable = { ZEX_MGY_royal_diet_temp_seats = temp }
	}
	else_if = {
		limit = {
			check_variable = { temp < 0 }
		}
		multiply_temp_variable = { temp = -1 }
		subtract_from_temp_variable = { ZEX_MGY_royal_diet_temp_seats = temp }
	}

	for_loop_effect = {
		start = 0
		end = ZEX_MGY_royal_diet_seat_array^num

		add_to_temp_array = { temp_array = v }
	}

	remove_from_temp_array = { array = temp_array value = temp_member }

	add_to_variable = { ZEX_MGY_royal_diet_seat_array^temp_member = ZEX_MGY_royal_diet_temp_seats }

	if = {
		limit = {
			check_variable = { ZEX_MGY_royal_diet_temp_seats > 0 }
		}
		while_loop_effect = {
			limit = {
				check_variable = { ZEX_MGY_royal_diet_temp_seats > 0 }
			}
			for_loop_effect = {
				start = 0
				end = temp_array^num

				if = {
					limit = {
						check_variable = { ZEX_MGY_royal_diet_seat_array^temp_array^v > 0 } # party must have more than 0 seats to be subtracted
						check_variable = { ZEX_MGY_royal_diet_temp_seats > 0 } # seats to be added is bigger than 0 currently
					}
					subtract_from_variable = { ZEX_MGY_royal_diet_seat_array^temp_array^v = 1 } # subtract from party 1 seat
					subtract_from_temp_variable = { ZEX_MGY_royal_diet_temp_seats = 1 } # subtract from seats to be added
				}
			}
		}
	}
	else_if = {
		limit = {
			check_variable = { ZEX_MGY_royal_diet_temp_seats < 0 }
		}
		while_loop_effect = {
			limit = {
				check_variable = { ZEX_MGY_royal_diet_temp_seats < 0 }
			}
			for_loop_effect = {
				start = 0
				end = temp_array^num

				if = {
					limit = {
						check_variable = { ZEX_MGY_royal_diet_seat_array^temp_array^v < parliament_main_array^num } # party must have less than max seats to be added
						check_variable = { ZEX_MGY_royal_diet_temp_seats < 0 } # seats to be subtracted is less than 0 currently
					}
					add_to_variable = { ZEX_MGY_royal_diet_seat_array^temp_array^v = 1 } # add to party 1 seat
					add_to_temp_variable = { ZEX_MGY_royal_diet_temp_seats = 1 } # add to seats to be subtracted
				}
			}
		}
	}
	ZEX_MGY_Diet_Reload_GUI = yes
}
