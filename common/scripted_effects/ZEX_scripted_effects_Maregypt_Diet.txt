# Start / Control
ZEX_MGY_Royal_Diet_Start_Diet = {
	set_country_flag = zex_mgy_royal_diet_ongoing
	ZEX_royal_diet_clear_gui_arrays = yes
	ZEX_MGY_Royal_Diet_Initialise_UI = yes
	ZEX_MGY_Define_Initial_Loyalty = yes
	ZEX_MGY_Calculate_Total_Support = yes
}

ZEX_MGY_Royal_Diet_end_Diet = {
	clr_country_flag = zex_mgy_royal_diet_ongoing
	ZEX_royal_diet_clear_gui_arrays = yes
}

# Set the initial amount of loyal seats
ZEX_MGY_Define_Initial_Loyalty = {
	set_variable = {
		ZEX_MGY_Royal_Diet_Zebrican_Loyalty = 3
	}
	set_variable = {
		ZEX_MGY_Royal_Diet_Harmonist_Loyalty = 45
	}
	set_variable = {
		ZEX_MGY_Royal_Diet_Loyalists_Loyalty = 12
	}
	set_variable = {
		ZEX_MGY_Royal_Diet_Independent_Loyalty = 5
	}
}

ZEX_MGY_Calculate_Total_Support = {
	#This is to reupdate the diet's support
	set_variable = {
		ZEX_MGY_Royal_Diet_Total_Support = 0
	}
	add_to_variable = {
		ZEX_MGY_Royal_Diet_Total_Support = ZEX_MGY_Royal_Diet_Zebrican_Loyalty
	}
	add_to_variable = {
		ZEX_MGY_Royal_Diet_Total_Support = ZEX_MGY_Royal_Diet_Harmonist_Loyalty
	}
	add_to_variable = {
		ZEX_MGY_Royal_Diet_Total_Support = ZEX_MGY_Royal_Diet_Loyalists_Loyalty
	}
	add_to_variable = {
		ZEX_MGY_Royal_Diet_Total_Support = ZEX_MGY_Royal_Diet_Independent_Loyalty
	}
}

ZEX_MGY_Clamp_Support_Values = {
	#Curbs overflow to avoid "invisible" supportive seats
	clamp_variable = {
		var = ZEX_MGY_Royal_Diet_Zebrican_Loyalty
		min = 0
		max = ZEX_MGY_royal_diet_seat_array^0
	}
	clamp_variable = {
		var = ZEX_MGY_Royal_Diet_Harmonist_Loyalty
		min = 0
		max = ZEX_MGY_royal_diet_seat_array^1
	}
	clamp_variable = {
		var = ZEX_MGY_Royal_Diet_Loyalists_Loyalty
		min = 0
		max = ZEX_MGY_royal_diet_seat_array^2
	}
	clamp_variable = {
		var = ZEX_MGY_Royal_Diet_Independent_Loyalty
		min = 0
		max = ZEX_MGY_royal_diet_seat_array^3
	}
}

#
# Effects to add / seats from each faction - 5, 10, 15, 20
# 
zex_mgy_add_zebrican_seats_5 = {
	custom_effect_tooltip = zex_mgy_add_zebrican_seats_5_tt
	hidden_effect = {
		#Sanity check to avoid negative numbers / overflow
		if = {
			limit = {
				check_variable = {
					ZEX_MGY_royal_diet_seat_array^0 > 195
				}
			}
			set_temp_variable = {
				var = ZEX_MGY_royal_diet_temp_seats
				value = 200
			}
			subtract_from_temp_variable = {
				var = ZEX_MGY_royal_diet_temp_seats
				value = ZEX_MGY_royal_diet_seat_array^0
			}
		}
		else = {
			set_temp_variable = {
				ZEX_MGY_royal_diet_temp_seats = 5
			}
		}
		# Actually add the seats
		set_temp_variable = {
			temp_member = 1
		}
		ZEX_MGY_Diet_modify_parliament_seat = yes
		ZEX_MGY_Clamp_Support_Values = yes
		ZEX_MGY_Clamp_Support_Values = yes
	}
}

	}	# Loyalists
	add_to_array = {

	}	# Independents
	ZEX_MGY_Diet_Get_Seats_Positions = yes
	ZEX_MGY_Diet_Reload_GUI = yes
}

ZEX_MGY_Diet_Reload_GUI = {
	clear_array = parliament_main_array
	for_each_loop = {
		array = ZEX_MGY_royal_diet_seat_array
		add_to_array = {
			temp_seat = ZEX_MGY_royal_diet_seat_array^i
		}
	}
	set_temp_variable = {
		temp_index_1 = 0
	}
	set_temp_variable = {
		temp_index_2 = 1
	}
	set_temp_variable = {
		temp_size = temp_seat^num
	}
	subtract_from_temp_variable = {
		temp_size = 1
	}
	while_loop_effect = {
		limit = {
			check_variable = {
				temp_index_1 < temp_size
			}
		}
		add_to_variable = {
			temp_seat^temp_index_2 = temp_seat^temp_index_1
		}
		add_to_temp_variable = {
			temp_index_1 = 1
		}
		add_to_temp_variable = {
			temp_index_2 = 1
		}
	}
	for_loop_effect = {
		start = 0
		end = temp_seat^num
		resize_array = {
			array = parliament_main_array
			value = parliament_member_array^v
			size = temp_seat^v
		}
	}
	clear_array = temp_seat
}

ZEX_MGY_Diet_modify_parliament_seat = {
	subtract_from_temp_variable = {
		temp_member = 1
	}
	set_temp_variable = {
		temp = ZEX_MGY_royal_diet_seat_array^temp_member
	}
	add_to_temp_variable = {
		temp = ZEX_MGY_royal_diet_temp_seats
	}
	if = {
		limit = {
			check_variable = {
				temp > parliament_main_array^num
			}
		}
		subtract_from_temp_variable = {
			temp = parliament_main_array^num
		}
		subtract_from_temp_variable = {
			ZEX_MGY_royal_diet_temp_seats = temp
		}
	}
	else_if = {
		limit = {
			check_variable = {
				temp < 0
			}
		}
		multiply_temp_variable = {
			temp = -1
		}
		subtract_from_temp_variable = {
			ZEX_MGY_royal_diet_temp_seats = temp
		}
	}
	for_loop_effect = {
		start = 0
		end = ZEX_MGY_royal_diet_seat_array^num
		add_to_temp_array = {
			temp_array = v
		}
	}
	remove_from_temp_array = {
		array = temp_array
		value = temp_member
	}
	add_to_variable = {
		ZEX_MGY_royal_diet_seat_array^temp_member = ZEX_MGY_royal_diet_temp_seats
	}
	if = {
		limit = {
			check_variable = {
				ZEX_MGY_royal_diet_temp_seats > 0
			}
		}
		while_loop_effect = {
			limit = {
				check_variable = {
					ZEX_MGY_royal_diet_temp_seats > 0
				}
			}
			for_loop_effect = {
				start = 0
				end = temp_array^num
				if = {
					limit = {
						check_variable = {
							ZEX_MGY_royal_diet_seat_array^temp_array^v > 0
						}						# party must have more than 0 seats to be subtracted
						check_variable = {
							ZEX_MGY_royal_diet_temp_seats > 0
						}						# seats to be added is bigger than 0 currently
					}
					subtract_from_variable = {
						ZEX_MGY_royal_diet_seat_array^temp_array^v = 1
					}					# subtract from party 1 seat
					subtract_from_temp_variable = {
						ZEX_MGY_royal_diet_temp_seats = 1
					}					# subtract from seats to be added
				}
			}
		}
	}
	else_if = {
		limit = {
			check_variable = {
				ZEX_MGY_royal_diet_temp_seats < 0
			}
		}
		while_loop_effect = {
			limit = {
				check_variable = {
					ZEX_MGY_royal_diet_temp_seats < 0
				}
			}
			for_loop_effect = {
				start = 0
				end = temp_array^num
				if = {
					limit = {
						check_variable = {
							ZEX_MGY_royal_diet_seat_array^temp_array^v < parliament_main_array^num
